613  strcpy(tmp_filename, tmp_filename_template);
614 
615 #ifdef WIN32
616  if (_mktemp(tmp_filename) == NULL)
617  {

619  }

621  unx_fd = mkstemp(tmp_filename);

624  return NULL;
625  }
626  close(unx_fd);
627 
628 #endif // #ifdef WIN32
629
